步骤:
        1.创建XMLHttpRequest对象
        2.配置请求方法和请求url地址
        3.监听loadend事件,接收响应结果
        4.发起请求

        const xhr = new XMLHttpRequest()
        xhr.open('请求方法','请求url网址')
        xhr.addEventlistener('loadend',()=>{
            //响应结果   
            console.log(xhr.response)
        })
        xhr.send()  //发送请求
