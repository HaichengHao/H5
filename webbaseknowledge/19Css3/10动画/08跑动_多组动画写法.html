多组动画写法,每组动画的复合属性用逗号分隔开

.outer{ width: 140px; height: 140px; overflow: hidden; /* margin: auto; */ background-image: url(./imgs/bg.png); /* 多组动画 */ animation: run_ 1s steps(12) 0s infinite normal forwards, toright 2s linear 0s 1 normal forwards ; } @keyframes run_{ from{ background-position: 0px 0px; } to{ background-position: -1680px 0px; } } @keyframes toright{ from{ transform: translateX(0); } to{ transform: translateX(980px); } }
