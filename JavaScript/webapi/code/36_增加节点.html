            const ol = document.querySelector('ol')
            const li1 = document.createElement('li')
            li.innerText = '2'
            ol.appendChild(li1)

            //上面都为在后面追加,接下来说一下在上面追加 insertBefore
            const li2 = document.createElement('li')
            li2.innerText = '0'
            ol.insertBefore(li2, li1)  //父元素.insertBefore(要插入的元素,在哪个元素之前)
